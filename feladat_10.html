10. Feladat
# be: pozitív egész # ki: a szám páros vagy páratlan

A program forrása:

        
szam = int(input("Kérek egy egész számot: "))
if szam % 2 == 0:
    print(f"A számod {szam} páros!")
if szam % 2 == 1:
    print(f"A számod {szam} páratlan!")

A program futtatása >>>

szam = int(input("Kérek egy egész számot: ")) if szam % 2 == 0: print(f"A számod {szam} páros!") if szam % 2 == 1: print(f"A számod {szam} páratlan!")
